import { useState, useRef, useEffect } from "react"
import { Search, Sparkles, BookOpen, Zap, Loader2, ArrowRight, Grid, LayoutList } from "lucide-react"
import { askQuestion } from "../lib/api"
import { useApi } from "../lib/useApi"

export function EcommerceRAGWorkspace() {
  const [messages, setMessages] = useState([])
  const [input, setInput] = useState("")
  const [useHybrid, setUseHybrid] = useState(true)
  const [viewMode, setViewMode] = useState("grid") // grid | list
  const { loading, error, execute } = useApi()
  const resultsEndRef = useRef(null)

  const handleSubmit = async (e) => {
    e.preventDefault()
    if (!input.trim() || loading) return

    const currentQuery = input.trim()
    setInput("")

    try {
      const res = await execute(() => askQuestion(currentQuery, useHybrid))
      
      // Store queries alongside their card payload
      setMessages((prev) => [
        ...prev,
        {
          id: Date.now().toString(),
          query: currentQuery,
          answer: res.answer,
          sources: res.sources || [],
          confidence: res.confidence,
          method: res.retrieval_method,
        },
      ])
    } catch {
      // Handled by useApi
    }
  }

  return (
    <div className="flex flex-col h-full min-h-screen bg-slate-50/60 font-sans antialiased text-slate-800">
      
      {/* ── Top Navigation Bar (E-Commerce Header Style) ── */}
      <header className="sticky top-0 z-50 bg-white border-b border-slate-200/80 px-6 py-4 shadow-[0_1px_2px_rgba(0,0,0,0.02)]">
        <div className="max-w-7xl mx-auto flex flex-col md:flex-row items-center justify-between gap-4">
          
          {/* Logo / Engine Label */}
          <div className="flex items-center gap-2.5 flex-shrink-0 self-start md:self-auto">
            <div className="w-8 h-8 rounded-lg bg-indigo-600 flex items-center justify-center text-white shadow-sm shadow-indigo-200">
              <Sparkles className="w-4 h-4" />
            </div>
            <div>
              <h1 className="text-sm font-bold tracking-tight text-slate-900 leading-none">CatalogAI</h1>
              <span className="text-[10px] text-slate-400 font-mono uppercase tracking-wider">Semantic Discovery</span>
            </div>
          </div>

          {/* ── Search Bar Chat Input ── */}
          <form onSubmit={handleSubmit} className="w-full max-w-2xl relative group">
            <div className="relative flex items-center bg-slate-100 focus-within:bg-white border border-transparent focus-within:border-indigo-500 rounded-xl transition-all duration-200 shadow-2xs focus-within:shadow-md">
              <Search className="absolute left-4 w-4 h-4 text-slate-400 group-focus-within:text-indigo-500 transition-colors" />
              <input
                type="text"
                value={input}
                onChange={(e) => setInput(e.target.value)}
                placeholder="Search the document ecosystem or ask a production question..."
                disabled={loading}
                className="w-full bg-transparent pl-11 pr-24 py-3 text-sm text-slate-800 placeholder:text-slate-400 focus:outline-none disabled:opacity-50"
              />
              <div className="absolute right-2 flex items-center gap-1.5">
                <button
                  type="submit"
                  disabled={!input.trim() || loading}
                  className="px-3 py-1.5 rounded-lg bg-indigo-600 hover:bg-indigo-700 disabled:opacity-0 disabled:scale-95 text-white text-xs font-medium transition-all duration-200 flex items-center gap-1 shadow-2xs"
                >
                  <span>Query</span>
                  <ArrowRight className="w-3 h-3" />
                </button>
              </div>
            </div>
          </form>

          {/* Settings & View Controls */}
          <div className="flex items-center gap-4 flex-shrink-0 self-end md:self-auto">
            <button
              type="button"
              onClick={() => setUseHybrid(!useHybrid)}
              className={`flex items-center gap-1.5 px-3 py-1.5 rounded-lg border text-xs font-medium transition-all ${
                useHybrid 
                  ? "bg-amber-50 border-amber-200 text-amber-700 shadow-2xs" 
                  : "bg-white border-slate-200 text-slate-500 hover:text-slate-700"
              }`}
            >
              <Zap className="w-3.5 h-3.5" />
              <span className="hidden sm:inline">Hybrid Routing</span>
            </button>

            <div className="h-5 w-[1px] bg-slate-200" />

            {/* Layout Toggles */}
            <div className="flex items-center bg-slate-100 p-0.5 rounded-lg border border-slate-200/40">
              <button 
                onClick={() => setViewMode("grid")}
                className={`p-1.5 rounded-md transition-all ${viewMode === "grid" ? "bg-white text-indigo-600 shadow-2xs" : "text-slate-400 hover:text-slate-600"}`}
              >
                <Grid className="w-3.5 h-3.5" />
              </button>
              <button 
                onClick={() => setViewMode("list")}
                className={`p-1.5 rounded-md transition-all ${viewMode === "list" ? "bg-white text-indigo-600 shadow-2xs" : "text-slate-400 hover:text-slate-600"}`}
              >
                <LayoutList className="w-3.5 h-3.5" />
              </button>
            </div>
          </div>

        </div>
      </header>

      {/* ── Main Catalog Result Canvas ── */}
      <main className="flex-1 max-w-7xl w-full mx-auto px-6 py-8 overflow-y-auto">
        
        {/* Empty State Showcase */}
        {messages.length === 0 && !loading && (
          <div className="flex flex-col items-center justify-center py-32 text-center animate-[fadeIn_0.4s_ease-out]">
            <div className="w-14 h-14 rounded-full bg-white border border-slate-200 flex items-center justify-center shadow-xs mb-4">
              <Search className="w-6 h-6 text-slate-300" />
            </div>
            <h2 className="text-base font-semibold text-slate-700">No active searches executed</h2>
            <p className="text-xs text-slate-400 mt-1 max-w-sm leading-relaxed">
              Type your context query in the navigation search layout above to dynamically isolate text clusters.
            </p>
          </div>
        )}

        {/* Global Result Streams */}
        <div className="space-y-12">
          {messages.map((item) => (
            <div key={item.id} className="space-y-4 animate-[slideUp_0.35s_ease-out]">
              
              {/* Search Result Query Summary Ribbon */}
              <div className="flex items-center justify-between border-b border-slate-200 pb-2">
                <div className="text-xs font-medium text-slate-400">
                  Search Query: <span className="text-slate-800 font-semibold font-mono bg-slate-100 px-2 py-0.5 rounded ml-1">"{item.query}"</span>
                </div>
                <div className="text-[11px] font-mono text-slate-400 uppercase tracking-wider">
                  {item.confidence} Accuracy Index · {item.method}
                </div>
              </div>

              {/* Central Synthesis Content */}
              <div className="bg-indigo-50/40 border border-indigo-100/60 rounded-xl p-5 shadow-2xs">
                <h4 className="text-[11px] font-bold uppercase tracking-widest text-indigo-600 mb-1">AI Synthesis Response</h4>
                <p className="text-[13.5px] leading-relaxed text-slate-700 font-serif">{item.answer}</p>
              </div>

              {/* ── The E-Commerce Product Cards Layout ── */}
              <div>
                <h4 className="text-[11px] font-bold uppercase tracking-widest text-slate-400 mb-3 px-1">Retrieved Document Matches</h4>
                
                <div className={
                  viewMode === "grid" 
                    ? "grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4" 
                    : "space-y-3"
                }>
                  {item.sources.map((src, i) => (
                    <div 
                      key={i} 
                      className="bg-white border border-slate-200 hover:border-indigo-400 rounded-xl p-4 shadow-2xs hover:shadow-md transition-all duration-200 flex flex-col justify-between group relative overflow-hidden"
                    >
                      {/* Top Meta Info (Brand Tag style) */}
                      <div>
                        <div className="flex items-center justify-between gap-2 mb-2">
                          <span className="text-[10px] font-bold font-mono text-indigo-600 uppercase bg-indigo-50 px-2 py-0.5 rounded">
                            Block #{i + 1}
                          </span>
                          <span className="text-[11px] font-medium text-slate-400 flex items-center gap-1">
                            <BookOpen className="w-3 h-3 text-slate-300" />
                            Page {src.page || "N/A"}
                          </span>
                        </div>

                        {/* Text Chunk Payload Container */}
                        <p className="text-xs leading-relaxed text-slate-600 font-sans line-clamp-4 group-hover:line-clamp-none transition-all duration-300">
                          {src.text || "Lorem ipsum dolor sit amet, consectetur adipiscing elit. Paragraph data segment pulled dynamically from the vector backend vector databases."}
                        </p>
                      </div>

                      {/* Footer Source Attribution */}
                      <div className="mt-4 pt-3 border-t border-slate-100 flex items-center justify-between text-[10px] font-medium text-slate-400">
                        <span className="truncate max-w-[180px]" title={src.source}>
                          {src.source.split("/").pop()}
                        </span>
                        <span className="text-indigo-500 font-semibold group-hover:underline cursor-pointer flex items-center gap-0.5">
                          Inspect Source →
                        </span>
                      </div>
                    </div>
                  ))}
                </div>
              </div>

            </div>
          ))}
        </div>

        {/* Global Catalog Loading Skeleton */}
        {loading && (
          <div className="mt-8 space-y-4 animate-pulse">
            <div className="h-4 w-48 bg-slate-200 rounded" />
            <div className="h-20 bg-slate-200/60 rounded-xl" />
            <div className="grid grid-cols-1 sm:grid-cols-3 gap-4 mt-4">
              {[1, 2, 3].map((n) => (
                <div key={n} className="h-40 bg-white border border-slate-200 rounded-xl p-4 space-y-3">
                  <div className="h-4 w-12 bg-slate-200 rounded" />
                  <div className="space-y-1.5">
                    <div className="h-3 w-full bg-slate-200 rounded" />
                    <div className="h-3 w-5/6 bg-slate-200 rounded" />
                    <div className="h-3 w-4/5 bg-slate-200 rounded" />
                  </div>
                </div>
              ))}
            </div>
          </div>
        )}

        <div ref={resultsEndRef} />
      </main>
    </div>
  )
}
